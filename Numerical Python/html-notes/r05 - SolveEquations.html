# Symboliczna specyfikacja problemu
p = sympy.symbols("p", positive=True)
A = sympy.Matrix([[1, sympy.sqrt(p)], [1, 1/sympy.sqrt(p)]])
b = sympy.Matrix([1, 2])

# Rozwizanie symboliczne
x_sym_sol = A.solve(b)
x_sym_sol.simplify()
Acond = A.condition_number().simplify()

# Funkcje do rozwizania numerycznego
AA = lambda p: np.array([[1, np.sqrt(p)], [1, 1/np.sqrt(p)]])
bb = np.array([1, 2])
x_num_sol = lambda p: np.linalg.solve(AA(p), bb)

# Wykrelenie rnic pomidzy symbolicznym (dokadnym) rozwizaniem a rozwizaniami numerycznymi
p_vec = np.linspace(0.9, 1.1, 200)

fig, axes = plt.subplots(1, 2, figsize=(12, 4))

style = ['r-', 'k--']
for n in range(2):
    x_sym = np.array([x_sym_sol[n].subs(p, pp).evalf() for pp in p_vec])
    x_num = np.array([x_num_sol(pp)[n] for pp in p_vec])
    axes[0].plot(p_vec, (x_num - x_sym)/x_sym, style[n])
axes[0].set_title("Bd rozwizania\n(numerycznesymboliczne)/symboliczne")
axes[0].set_xlabel(r'$p$', fontsize=18)

axes[1].plot(p_vec, [Acond.subs(p, pp).evalf() for pp in p_vec])
axes[1].set_title("Wspczynnik uwarunkowania")
axes[1].set_xlabel(r'$p$', fontsize=18)

fig.tight_layout()

np.random.seed(1234)

# Rzeczywiste parametry modelu
x = np.linspace(-1, 1, 100)
a, b, c = 1, 2, 3
y_exact = a + b * x + c * x**2

# Symulacja zaszumionych danych
m = 100
X = 1 - 2 * np.random.rand(m)
Y = a + b * X + c * X**2 + np.random.randn(m)

# Dopasowanie modelu do danych z uyciem metody najmniejszych kwadratw
A = np.vstack([X**0, X**1, X**2])  # Zamiast vstack mona rwnie wykorzysta np.vander
A = A.T

sol, r, rank, sv = la.lstsq(A, Y)
y_fit = sol[0] + sol[1] * x + sol[2] * x**2

fig, ax = plt.subplots(figsize=(12, 4))
ax.plot(X, Y, 'go', alpha=0.5, label='Dane z symulacji')
ax.plot(x, y_exact, 'k', lw=2, label='$y = 1 + 2x + 3x^2$ (prawdziwe wartoci)')
ax.plot(x, y_fit, 'b', lw=2, label='Dopasowanie metod najmniejszych kwadratw')
ax.set_xlabel("$x$", fontsize=18)
ax.set_ylabel("$y$", fontsize=18, rotation=0)
ax.legend(loc=2);

# Definicja funkcji, okrelenie dokadnoci i przedziau pocztkowego [a, b]
x = np.linspace(-2.1, 2.1, 1000)
f = lambda x: np.exp(x) - 2
tol = 0.1
a, b = -2, 2


# Wykres funkcji f
fig, ax = plt.subplots(1, 1, figsize=(12, 4))

ax.plot(x, f(x), lw=1.5)
ax.axhline(0, ls=':', color='k')
ax.set_xticks([-2, -1, 0, 1, 2])
ax.set_xlabel(r'$x$', fontsize=18)
ax.set_ylabel(r'$f(x)$', fontsize=18)

# Poszukiwanie pierwiastka metod bisekcji wraz z 
# zaznaczaniem kolejnych krokw na wykresie
fa, fb = f(a), f(b)

ax.plot(a, fa, 'ko')
ax.plot(b, fb, 'ko')
ax.text(a, fa + 0.5, r"$a$", ha='center', fontsize=18)
ax.text(b, fb + 0.5, r"$b$", ha='center', fontsize=18)

n = 1
while b - a > tol:
    m = a + (b - a)/2
    fm = f(m)

    ax.plot(m, fm, 'ko')
    ax.text(m, fm - 0.5, f"$m_{n}$", ha='center')
    n += 1
    
    if np.sign(fa) == np.sign(fm):
        a, fa = m, fm
    else:
        b, fb = m, fm

ax.plot(m, fm, 'r*', markersize=10)
ax.annotate("Pierwiastek rwny okoo %.3f" % m,
            fontsize=14, family="serif",
            xy=(a, fm), xycoords='data',
            xytext=(-150, +50), textcoords='offset points', 
            arrowprops=dict(arrowstyle="->", connectionstyle="arc3, rad=-.5"))

ax.set_title("Metoda bisekcji")

# Definicja funkcji, okrelenie dokadnoci i punktu startowego xk
tol = 0.01
xk = 2

s_x = sympy.symbols("x")
s_f = sympy.exp(s_x) - 2

f = lambda x: sympy.lambdify(s_x, s_f, 'numpy')(x)
fp = lambda x: sympy.lambdify(s_x, sympy.diff(s_f, s_x), 'numpy')(x)

x = np.linspace(-1, 2.1, 1000)

# Tworzenie wykresu do wizualizacji kolejnych krokw
fig, ax = plt.subplots(1, 1, figsize=(12,4))

ax.plot(x, f(x))
ax.axhline(0, ls=':', color='k')

# Metoda Newtona jest powtarzana, a do osignicia zadanej dokadnoci
n = 0
while f(xk) > tol:
    xk_new = xk - f(xk) / fp(xk)

    ax.plot([xk, xk], [0, f(xk)], color='k', ls=':')
    ax.plot(xk, f(xk), 'ko')
    ax.text(xk, -.5, r'$x_%d$' % n, ha='center')
    ax.plot([xk, xk_new], [f(xk), 0], 'k-')

    xk = xk_new
    n += 1

ax.plot(xk, f(xk), 'r*', markersize=15)
ax.annotate("Pierwiastek rwny okoo %.3f" % xk,
            fontsize=14, family="serif",
            xy=(xk, f(xk)), xycoords='data',
            xytext=(-150, +50), textcoords='offset points', 
            arrowprops=dict(arrowstyle="->", connectionstyle="arc3, rad=-.5"))

ax.set_title("Metoda Newtona")
ax.set_xticks([-1, 0, 1, 2])
fig.tight_layout()
